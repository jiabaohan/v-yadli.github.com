            unsigned CurrentTurn;
            unsigned ArrangedTurn;
        }

        void GetLock(SpinLock* lock)
        {
            unsigned myTurn = InterlockedIncrement(&lock->ArrangedTurn);
            while(*&lock->CurrentTurn != myTurn)
                __mm_pause();
        }

        void UnLock(SpinLock* lock)

            ++lock->CurrentTurn;
